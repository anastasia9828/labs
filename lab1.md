## 1.	Створити змінні базових (atomic) типів. Базові типи: character, numeric, integer, complex, logical. ##
character <- "Приклад" <br>
numeric <- 8 <br>
integer <- 8L <br>
complex <- 8 + 5i <br>
logical <- TRUE <br>
> character <- "Приклад" <br>
> numeric <- 8 <br>
> integer <- 8L <br>
> complex <- 8 + 5i <br>
> logical <- TRUE <br>
